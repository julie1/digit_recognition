def swap(labels):
    i1 = np.argwhere(labels == 10)
    i2 = np.argwhere(labels == 0)
    for i in i1:labels[i] = 0
    for j in i2:labels[j] = 10

"""
Compute Intersection-over-Union of 2 bounding boxes
in the form b = [y0, x0, y1, x1]
"""

def IOU_loss(b1, b2):
    yi0 = max(b1[0], b2[0])
    xi0 = max(b1[1], b2[1])
    yi1 = min((b1[2], b2[2]))
    xi1 = min(b1[3], b2[3])
    b1area = (b1[3] - b1[1])*(b1[2] - b1[0])
    b2area = (b2[3] - b2[1])*(b2[2] - b2[0])
    biarea = (xi1 - xi0)*(yi1 - yi0)
    if xi0>xi1 or yi0>yi1:return 0
    else:return 1.0 - float(biarea/(b1area+b2area-biarea))

def fill_feed_dict(data, labels, x, y_, z, step, batch_size, train):
    size = labels.shape[0]
    # Compute the offset of the current minibatch in the data.
    # Note that we could use better randomization across epochs.
    offset = (step * batch_size) % (size - batch_size)
    batch_data = data[offset:(offset + batch_size), ...]
    batch_labels = labels[offset:(offset + batch_size)]
    return {x: batch_data, y_: batch_labels, z : train }

    # Create model
    # Utility functions
    def weight_variable(shape, name):
      initial = tf.truncated_normal(shape, stddev=0.1)
      return tf.Variable(initial, name=name)


    def bias_variable(b, shape, name):
      initial = tf.constant(b, shape=shape)
      return tf.Variable(initial, name=name)


    def conv2d(x, W, stride=(1, 1), padding='VALID'):
      return tf.nn.conv2d(x, W, strides=[1, stride[0], stride[1], 1],
                          padding=padding)


    def max_pool(x, ksize=(2, 2), stride=(2, 2)):
      return tf.nn.max_pool(x, ksize=[1, ksize[0], ksize[1], 1],
                            strides=[1, stride[0], stride[1], 1], padding='SAME')


    def avg_pool(x, ksize=(2, 2), stride=(2, 2)):
      return tf.nn.avg_pool(x, ksize=[1, ksize[0], ksize[1], 1],
                            strides=[1, stride[0], stride[1], 1], padding='SAME')


    def convolutional_layers(x):
        """
        Get the convolutional layers of the model.
        """
       # x = tf.placeholder(tf.float32, [None, None, None, 1])

        # First layer
        W_conv1 = weight_variable([5, 5, 1, 48], "W_conv1")
        b_conv1 = bias_variable(0.0, [48], "b_conv1")
        
    
        h_conv1 = tf.nn.relu(conv2d(x, W_conv1) + b_conv1)
        h_pool1 = max_pool(h_conv1, ksize=(2, 2), stride=(2, 2))
        lrn = tf.nn.local_response_normalization(h_pool1, name="Normalize_1")


        # Second layer
        W_conv2 = weight_variable([5, 5, 48, 64], "W_conv2")
        b_conv2 = bias_variable(0.1, [64], "b_conv2")

        h_conv2 = tf.nn.relu(conv2d(h_pool1, W_conv2) + b_conv2)
        h_pool2 = max_pool(h_conv2, ksize=(2, 2), stride=(2, 2))
        lrn = tf.nn.local_response_normalization(h_pool2, name="Normalize_2")


        # Third layer
        W_conv3 = weight_variable([5, 5, 64, 128], "W_conv3")
        b_conv3 = bias_variable(0.1, [128], "b_conv3")

        h_conv3 = tf.nn.relu(conv2d(h_pool2, W_conv3) + b_conv3)
        lrn = tf.nn.local_response_normalization(h_conv3)

        if lrn.get_shape().as_list()[1] is 1:  # Is already reduced.
            h_pool3 = max_pool(lrn, ksize=(1, 1), stride=(1, 1)) 
                                 
        else:
            h_pool3 = max_pool(lrn, ksize=(2, 2), stride=(2, 2))
                                 

        
      #  regularizer = np.multiply(p, tf.nn.l2_loss(W_conv3))

        return h_pool3, [W_conv1, b_conv1,
                            W_conv2, b_conv2,
                            W_conv3, b_conv3]


    
    
    def get_training_model(x, keep_prob=1.0):
        """
        The training model acts on a batch of 32x128 windows, and outputs a (1 +
        5 * len(DIGIT) vector, `v`. `v[0]` is the probability that a plate is
        fully within the image and is at the correct scale.
    
        `v[1 + i * len(DIGITS) + c]` is the probability that the `i`'th
        character is `c`.
        """
        conv_layer, conv_vars = convolutional_layers(x)
        shape = tf.shape(conv_layer)
       # dim = tf.multiply(tf.multiply(shape[1], shape[2]), shape[3])

        if train==1:
            print("Using drop out")
            conv_layer = tf.nn.dropout(conv_layer, DROPOUT)
        else:
            print("Not using dropout")
      
        
    
        # Densely connected layer
        W_fc1 = weight_variable([128, 160], "W_fc1")
        b_fc1 = bias_variable(0.05, [160], "b_fc1")

        conv_layer_flat = tf.reshape(conv_layer, [shape[0], -1])
        h_fc1 = tf.nn.relu(tf.matmul(conv_layer_flat, W_fc1) + b_fc1)

        # Output layer
        


        W_fc2 = weight_variable([160, len(DIGITS)], "W_fc2") 
        b_fc2  = bias_variable(0.05, [len(DIGITS)], "b_fc2") 
        y = tf.matmul(h_fc1, W_fc2) + b_fc2 
        
        return (y, conv_vars + [W_fc1, b_fc1, W_fc2, b_fc2])
    
    
    def multi_digits(x, train=0):
        conv_layer, conv_vars = convolutional_layers(x)
    
        # Fourth layer
        W_conv4 = weight_variable([5, 5, 128, 160], "W_conv4")        
        b_conv4 = bias_variable(0.1, [160], "b_conv4")
        h_conv4 = tf.nn.relu(conv2d(conv_layer, W_conv4,
                                stride=(2, 2)) + b_conv4) 
        # Output layer
        W_out = [weight_variable([160, len(DIGITS)], "W_out"+str(i)) for i in range(5)]       
        b_out = [bias_variable(1.0, [len(DIGITS)], "b_out"+str(i)) for i in range(5)]
        shape = tf.shape(h_conv4)
        h_conv4_flat = tf.reshape(h_conv4, [shape[0], -1])
        y = [tf.matmul(h_conv4_flat, W_out[i]) + b_out[i] for i in range(5)]        
        y = tf.pack(y, axis=1)
        return ( y, conv_vars + [W_conv4, b_conv4, W_out, b_out])
