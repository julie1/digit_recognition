image_sizex = 32  # Pixel width 
image_sizey = 32  # Pixel height.
pixel_depth = 255.0  # Number of levels per pixel.
num_channels = 1


def make_image_array(im, boxes):
    
    digits = []
    for box in boxes:        
        y0, x0, y1, x1 = box[0], box[1], box[2], box[3]
      #  print(y0,y1,x0,x1)
        if y1 > y0 and x1 > x0:
            
            x = x1 - x0
            y = y1 - y0
         #   print(x,y)
            
            box_left = int(max((np.floor(x0 - 0.1 * x), 0)))
            box_top = int(max(np.floor(y0 - 0.1 * y), 0))
            
            box_right = int(np.ceil(np.amin([np.ceil(box_left + 1.2 * x), im.shape[1]])))
            box_bottom = int(np.ceil(np.amin([np.ceil(y0 + 1.2 * y), im.shape[0]])))
          #  print(box_top, box_bottom, box_left, box_right)
            x2 = box_right - box_left
            y2 = box_bottom - box_top
         #   print(x2,y2)
            
            new_im = im[box_top:box_bottom, box_left:box_right, :]
         #   print(new_im.shape)
            new_im = ndimage.interpolation.zoom(new_im, [32./y2, 32./x2, 1])
           # print(new_im.shape)
            norm_im = (new_im - pixel_depth / 2) / pixel_depth
           # norm_im = np.reshape(new_im, (image_sizey, image_sizex, num_channels))
            digits.append(norm_im)
            
    return digits

    dataset = np.ndarray(shape=(data.shape[0]*5, image_sizey, image_sizex, num_channels),
                         dtype=np.float32)

          #  if digits[i].shape[0]  ==0:print(i,l, index/5)
            dataset[index+i, :, :, :] = digits[i]
        
        if l < 5:
            for j in range(5-l):
                dataset[index+l+j, :, :] = 0
        index += 5
    dataset = dataset[0:index, ...]
    labels = np.reshape(lab, (dataset.shape[0]))

def fill_feed_dict(data, labels, x, y_, z, step, batch_size, train):
    size = labels.shape[0]
    # Compute the offset of the current minibatch in the data.
    # Note that we could use better randomization across epochs.
    offset = (step * batch_size) % (size - batch_size)
    batch_data = data[offset:(offset + batch_size), ...]
    batch_labels = labels[offset:(offset + batch_size)]
    return {x: batch_data, y_: batch_labels, z : train }

    def weight_variable(shape, name):
      initial = tf.truncated_normal(shape, stddev=0.1)
      return tf.Variable(initial, name=name)


    def bias_variable(b, shape, name):
      initial = tf.constant(b, shape=shape)
      return tf.Variable(initial, name=name)


    def conv2d(x, W, stride=(1, 1), padding='VALID'):
      return tf.nn.conv2d(x, W, strides=[1, stride[0], stride[1], 1],
                          padding=padding)


    def max_pool(x, ksize=(2, 2), stride=(2, 2)):
      return tf.nn.max_pool(x, ksize=[1, ksize[0], ksize[1], 1],
                            strides=[1, stride[0], stride[1], 1], padding='SAME')

    def convolutional_layers(x):
        """
        Get the convolutional layers of the model.
        """
       # x = tf.placeholder(tf.float32, [None, None, None, 1])

        # First layer
        W_conv1 = weight_variable([5, 5, 1, 48], "W_conv1")
        b_conv1 = bias_variable(0.0, [48], "b_conv1")
        
    
        h_conv1 = tf.nn.relu(conv2d(x, W_conv1) + b_conv1)
        h_pool1 = max_pool(h_conv1, ksize=(2, 2), stride=(2, 2))
        lrn = tf.nn.local_response_normalization(h_pool1, name="Normalize_1")


        # Second layer
        W_conv2 = weight_variable([5, 5, 48, 64], "W_conv2")
        b_conv2 = bias_variable(0.1, [64], "b_conv2")

        h_conv2 = tf.nn.relu(conv2d(h_pool1, W_conv2) + b_conv2)
        h_pool2 = max_pool(h_conv2, ksize=(2, 2), stride=(2, 2))
        lrn = tf.nn.local_response_normalization(h_pool2, name="Normalize_2")


        # Third layer
        W_conv3 = weight_variable([5, 5, 64, 128], "W_conv3")
        b_conv3 = bias_variable(0.1, [128], "b_conv3")

        h_conv3 = tf.nn.relu(conv2d(h_pool2, W_conv3) + b_conv3)
        lrn = tf.nn.local_response_normalization(h_conv3)

        if lrn.get_shape().as_list()[1] is 1:  # Is already reduced.
            h_pool3 = max_pool(lrn, ksize=(1, 1), stride=(1, 1)) 
                                 
        else:
            h_pool3 = max_pool(lrn, ksize=(2, 2), stride=(2, 2))
                                 

        
      #  regularizer = np.multiply(p, tf.nn.l2_loss(W_conv3))

        The training model acts on a batch of 32x128 windows, and outputs a (1 +

        fully within the image and is at the correct scale.
    
        `v[1 + i * len(DIGITS) + c]` is the probability that the `i`'th
        character is `c`.

        conv_layer, conv_vars = convolutional_layers(x)

        W_fc1 = weight_variable([128, 160], "W_fc1")
        b_fc1 = bias_variable(0.05, [160], "b_fc1")

        conv_layer_flat = tf.reshape(conv_layer, [shape[0], -1])
        h_fc1 = tf.nn.relu(tf.matmul(conv_layer_flat, W_fc1) + b_fc1)

        # Output layer
        


        W_fc2 = weight_variable([160, len(DIGITS)], "W_fc2") 
        b_fc2  = bias_variable(0.05, [len(DIGITS)], "b_fc2") 
        y = tf.matmul(h_fc1, W_fc2) + b_fc2 
        
        return (y, conv_vars + [W_fc1, b_fc1, W_fc2, b_fc2])

        y = tf.pack(y, axis=1)
        return ( y, conv_vars + [W_conv4, b_conv4, W_out, b_out])

    tf_multidigits = tf.placeholder(tf.float32, shape=(batch_size, n_input0))
    tf_multi_labels_hot = tf.placeholder(tf.float32, shape=(batch_size, n_sequence, n_classes))
    is_train = tf.placeholder(tf.int32)
    
    # Create model
    def multilayer_perceptron(_X, _weights, _biases):
        layer_1 = tf.nn.relu(tf.add(tf.matmul(_X, _weights['h1']), _biases['b1'])) #Hidden layer with RELU activation
        layer_2 = tf.nn.relu(tf.add(tf.matmul(layer_1, _weights['h2']), _biases['b2'])) #Hidden layer with RELU activation
        
        return tf.matmul(layer_2, _weights['out']) + _biases['out']
    
    def multidigit_perceptron(_X, _weights, _biases):
        layer_0 = tf.nn.relu(tf.add(tf.matmul(_X, _weights['h0']), _biases['b0'])) #Hidden layer with RELU activation
        layer_01 = tf.nn.relu(tf.add(tf.matmul(layer_0, _weights['h01']), _biases['b01'])) #Hidden layer with RELU activation1        
        layer_1 = tf.nn.relu(tf.add(tf.matmul(layer_01, _weights['h1']), _biases['b1'])) #Hidden layer with RELU activation
        layer_2 = tf.nn.relu(tf.add(tf.matmul(layer_1, _weights['h2']), _biases['b2'])) #Hidden layer with RELU activation
        out = [tf.matmul(layer_2, _weights['out1'][i]) + _biases['out1'][i] for i in range(5)]
        return tf.pack(out, axis=1)
        

    # Store layers weight & bias
    weights = {
        'h0': tf.Variable(tf.truncated_normal([n_input0, n_hidden_0], stddev = 0.05)),
        'h01': tf.Variable(tf.truncated_normal([n_hidden_0, n_input], stddev = 0.05)),
        'h1': tf.Variable(tf.truncated_normal([n_input, n_hidden_1], stddev = 0.05)),
        'h2': tf.Variable(tf.truncated_normal([n_hidden_1, n_hidden_2], stddev = 0.05)),
        'out': tf.Variable(tf.truncated_normal([n_hidden_2, n_classes], stddev = 0.05)),
        'out1':[tf.Variable(tf.truncated_normal([n_hidden_2, n_classes], stddev = 0.05)) for i in range(5)]
    }
    biases = {
        'b0': tf.Variable(tf.zeros([n_hidden_0])),
        'b01': tf.Variable(tf.zeros([n_input])),
        'b1': tf.Variable(tf.zeros([n_hidden_1])),
        'b2': tf.Variable(tf.zeros([n_hidden_2])),
        'out': tf.Variable(tf.zeros([n_classes])),
        'out1': [tf.Variable(tf.zeros([n_classes])) for i in range(5)]
    }

    learning_rate = tf.train.exponential_decay(0.4, global_step, 1000, 0.65, staircase=True)
